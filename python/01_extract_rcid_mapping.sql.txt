WITH active_companies AS (
  SELECT DISTINCT p.rcid
  FROM revelio_job_postings_old.postings_linkedin p
  WHERE p.country = 'United States'
    AND p.post_date >= DATE '2024-01-01'
    AND p.post_date <  DATE '2024-02-01'
    AND p.rcid IS NOT null
)
SELECT DISTINCT ON (c.rcid)
  c.rcid,
  c.company,
  c.url,
  c.naics_code,
  c.year_founded
FROM revelio_common.company_mapping c
JOIN active_companies a
  ON a.rcid = c.rcid
ORDER BY c.rcid, c.url NULLS LAST;
