WITH lobby_to_rcid (clientship_id, rcid) AS (
  values

),
ai_firms AS (
  SELECT DISTINCT x.clientship_id
  FROM lobby_to_rcid x
  JOIN revelio_job_postings.postings_cosmos p
    ON p.rcid = x.rcid
    AND p.country = 'United States'
  	AND p.source_linkedin = TRUE
    AND p.post_date >= DATE '2023-01-01'
    AND p.post_date <  DATE '2023-01-31'
  JOIN revelio_job_postings.postings_cosmos_raw r
    ON r.job_id = p.job_id
  where (
 	 r.description ILIKE '%artificial intelligence%'
 	 OR r.description ILIKE '%machine learning%'
 	 OR r.description ILIKE '%deep learning%'
 	 OR r.description ILIKE '%generative ai%'
 	 OR r.description ILIKE '%large language model%'
 	 OR r.description ILIKE '%llm%'
 	 OR r.description ILIKE '%natural language processing%'
 	 OR r.description ILIKE '%nlp%'
 	 OR r.description ILIKE '%computer vision%'
 	 OR r.description ILIKE '%reinforcement learning%'
 	 OR r.description ILIKE '%neural network%'
 	 OR r.description ILIKE '%transformer%'
 	 OR r.description ILIKE '%foundation model%'
 	 OR r.description ILIKE '%tensorflow%'
 	 OR r.description ILIKE '%pytorch%'
	 OR r.description ILIKE '%hugging face%'
     )
)
SELECT
  x.clientship_id,
  CASE WHEN a.clientship_id IS NOT NULL THEN 1 ELSE 0 END AS ai_flag_2023_01
FROM lobby_to_rcid x
LEFT JOIN ai_firms a
  ON a.clientship_id = x.clientship_id;
